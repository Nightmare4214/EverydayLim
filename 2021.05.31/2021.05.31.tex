\begin{mybox}{问题2021.05.31}
	
	\qquad 设$A,B$均是$n$阶方阵,已知$A-E$可逆,$\left|B\right|=1$,且$(A-E)^{-1}=B^{*}-E$,其中$B^{*}$为$B$的伴随矩阵,则$A^{-1}=$\underline{\hbox to 20mm{}}
\end{mybox}	
\noindent
\textbf{解：}
\begin{align*}
	(A-E)^{-1}&=B^{*}-E\\
	(A-E)^{-1}&=\left|B\right| B^{-1} -BB^{-1}\\
	(A-E)^{-1}&=(E-B)B^{-1}\\
	A-E&=B(E-B)^{-1}\\
	A&=B(E-B)^{-1}+E\\
	A&=B(E-B)^{-1}+(E-B)(E-B)^{-1}\\
	A&=(E-B)^{-1}\\
	A^{-1}&=E-B
\end{align*}
