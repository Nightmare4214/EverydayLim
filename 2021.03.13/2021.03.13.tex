\begin{mybox}{问题2021.03.13}
	
	\qquad $\lim\limits_{x \to 0^+} x^{\ln(\frac{\ln x-1}{\ln x+1})}$
\end{mybox}
\noindent
\textbf{解}
\begin{align*}
	&\quad\lim\limits_{x \to 0^+} x^{\ln(\frac{\ln x-1}{\ln x+1})}\\
	&=\lim\limits_{x \to 0^+} x^{-\frac{2}{\ln x+1}}\\
	&=e^{\lim\limits_{x\to 0^+}\frac{-2\ln x}{\ln x+1}}\\
	&=e^{-2}
\end{align*}
